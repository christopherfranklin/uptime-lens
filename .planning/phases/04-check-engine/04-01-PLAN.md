   ```typescript
   import { neon } from "@neondatabase/serverless";
   import { drizzle } from "drizzle-orm/neon-http";
   import type { NeonHttpDatabase } from "drizzle-orm/neon-http";
   import * as schema from "../../lib/db/schema";

   export type WorkerDb = NeonHttpDatabase<typeof schema>;

   export function createWorkerDb(): WorkerDb {
     const sql = neon(process.env.DATABASE_URL!);
     return drizzle({ client: sql, schema });
   }
   ```

    - Calls socket.destroy() on every code path (connect, timeout, error)
    - Never rejects -- always resolves with ProbeResult

2. Create `worker/src/probes/http.ts` -- export `probeHttp(url: string, timeoutMs: number, expectedStatusCode: number): Promise<ProbeResult>`:
   - Use built-in `fetch` with `redirect: 'manual'` and `AbortSignal.timeout(timeoutMs)`
   - Measure time with `performance.now()`
   - Success = response.status === expectedStatusCode
   - On catch: detect timeout (message includes 'TimeoutError' or 'abort') vs other errors
   - ALWAYS resolve, never reject

3. Create `worker/src/probes/tcp.ts` -- export `probeTcp(host: string, port: number, timeoutMs: number): Promise<ProbeResult>`:
   - Use `net.createConnection({ host, port, timeout: timeoutMs })`
   - On 'connect': record time, call socket.destroy(), resolve success
   - On 'timeout': socket.destroy(), resolve failure
   - On 'error': socket.destroy(), resolve failure
   - ALWAYS resolve, never reject

4. Create `worker/src/probes/ssl.ts` -- export `probeSsl(host: string, port: number, timeoutMs: number): Promise<ProbeResult>`:
   - Use `tls.connect({ host, port, servername: host, timeout: timeoutMs, rejectUnauthorized: false })`
   - On 'secureConnect': getPeerCertificate(), extract valid_to -> sslExpiresAt
   - On 'timeout': socket.destroy(), resolve failure
   - On 'error': socket.destroy(), resolve failure
   - If cert missing or no valid_to: resolve failure with "No certificate returned"
   - ALWAYS resolve, never reject
