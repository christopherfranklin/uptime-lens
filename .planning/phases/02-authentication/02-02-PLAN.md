1. Create `proxy.ts` at project root (NOT `middleware.ts` -- Next.js 16 uses `proxy.ts` per Pitfall #2):
   ```typescript
   import { NextRequest, NextResponse } from "next/server";
   import { getSessionCookie } from "better-auth/cookies";

   const protectedRoutes = ["/dashboard", "/settings"];
   const authRoutes = ["/login"];

   export function proxy(request: NextRequest) {
     const path = request.nextUrl.pathname;
     const sessionCookie = getSessionCookie(request);

     // Redirect unauthenticated users to login
     if (protectedRoutes.some((route) => path.startsWith(route)) && !sessionCookie) {
       return NextResponse.redirect(new URL("/login", request.url));
     }

     // Redirect authenticated users away from auth pages to dashboard
     if (authRoutes.some((route) => path.startsWith(route)) && sessionCookie) {
       return NextResponse.redirect(new URL("/dashboard", request.url));
     }

     return NextResponse.next();
   }

   export const config = {
     matcher: ["/((?!api|_next/static|_next/image|favicon\\.ico|.*\\.png$).*)"],
   };
   ```
   - This is an OPTIMISTIC check only (cookie existence, no DB call) -- per research anti-pattern guidance.
   - No return URL tracking per user decision.

2. Create `lib/dal.ts` -- Data Access Layer for secure session validation:
   ```typescript
   import "server-only";
   import { cache } from "react";
   import { headers } from "next/headers";
   import { redirect } from "next/navigation";
   import { auth } from "@/lib/auth";

   export const verifySession = cache(async () => {
     const session = await auth.api.getSession({
       headers: await headers(),
     });
     if (!session) {
       redirect("/login");
     }
     return session;
   });

   export const getOptionalSession = cache(async () => {
     return auth.api.getSession({
       headers: await headers(),
     });
   });
   ```
   - `verifySession()`: use in pages that REQUIRE auth (dashboard, settings). Redirects to /login if no session.
   - `getOptionalSession()`: use in pages that optionally show user state (e.g., marketing header showing user name).
   - Both are memoized with `cache()` -- multiple calls in one request hit DB only once.
   - Install `server-only` package if not already present: `npm install server-only`.

Run `npm run lint` to ensure Biome compliance.
